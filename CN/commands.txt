ipconfig                # Shows the IP configuration of your machine.
ipconfig /all           # displays full detailed network configuration (including MAC address, DHCP, DNS info).
ping google.com         # Tests connectivity to another host (like google.com).
tracert google.com      # Shows the path taken to reach a destination host.
netstat                 # Displays active connections and listening ports.
netstat -ano            # Shows active connections with process IDs.
nslookup google.com     # Checks DNS resolution for a domain.
arp -a                  # Displays the ARP (Address Resolution Protocol) table.
route print             # Displays the routing table for your system.
hostname                # Displays the name of your computer (host).
